The system of equations have a total of 21 nonlinear variables, organized into four categories. Five equations are needed to describe the fluid motion of neutral particles.
\begin{align}
    \frac{\partial \rho}{\partial t} + \nabla \cdot \rho \mathbf{u} &= \frac{M}{N_A} (\Gamma_{\text{rec}} - \Gamma_{\text{pi}} - \Gamma_{\text{ci}}) \\
    \frac{\partial \rho\mathbf{u}}{\partial t} + \nabla \cdot \rho \mathbf{u} \otimes \mathbf{u} &= -\nabla p + \nabla \cdot \mathbf{T} + \rho\mathbf{g} + \frac{M}{N_A} [\mathbf{f}_{en} + \mathbf{f}_{in} + \Gamma_{\text{rec}}\mathbf{u}_i - (\Gamma_{\text{pi}} + \Gamma_{\text{ci}})\mathbf{u} ] \\
    \frac{\partial \rho E}{\partial t} + \nabla \cdot \rho\mathbf{u} H &= -\nabla \cdot \mathbf{q}'' + \nabla \cdot \left[\mathbf{T} \cdot \mathbf{u} \right] + q'''_{\text{exc}} + q'''_{en} + q'''_{in} + \frac{M}{N_A} [\Gamma_{\text{rec}}E_i - (\Gamma_{\text{pi}} + \Gamma_{\text{ci}})E]
\end{align}

Electron motion requires another set of five equations.
\begin{align}
    \frac{\partial n_e}{\partial t} + \nabla \cdot n_e \mathbf{u}_e &= \Gamma_{\text{pi}} + \Gamma_{\text{ci}} - \Gamma_{\text{rec}} \\
    \frac{\partial n_e\mathbf{u}_e}{\partial t} + \nabla \cdot n_e \mathbf{u}_e \otimes \mathbf{u}_e &= \frac{-\nabla p_e - en_e (\mathbf{E} + \mathbf{u}_e \times \mathbf{B})}{m_e} - \mathbf{f}_{en} \\
    \frac{\partial n_e \mathcal{E}_e}{\partial t} + \nabla \cdot n_e \mathbf{u}_e \mathcal{H}_e &= q'''_{\text{pi}} - q'''_{en} - q'''_{ei} - en_e\mathbf{u}_e \cdot \mathbf{E}
\end{align}

Ion motion is described by a very similar set of equations.
\begin{align}
    \frac{\partial n_i}{\partial t} + \nabla \cdot n_i \mathbf{u}_i &= \Gamma_{\text{pi}} + \Gamma_{\text{ci}} - \Gamma_{\text{rec}} \\
    \frac{\partial n_i\mathbf{u}_i}{\partial t} + \nabla \cdot n_i \mathbf{u}_i \otimes \mathbf{u}_i &= \frac{-\nabla p_i + en_i (\mathbf{E} + \mathbf{u}_i \times \mathbf{B}) }{m_i} - \mathbf{f}_{in} + (\Gamma_{\text{pi}} + \Gamma_{\text{ci}})\mathbf{u} - \Gamma_{\text{rec}}\mathbf{u}_i \\
    \frac{\partial n_i \mathcal{E}_i}{\partial t} + \nabla \cdot n_i \mathbf{u}_i \mathcal{H}_i &= q'''_{ei} - q'''_{in} + en_i\mathbf{u}_i \cdot \mathbf{E} + (\Gamma_{\text{pi}} + \Gamma_{\text{ci}}) \mathcal{E} - \Gamma_{\text{rec}}\mathcal{E}_i
\end{align}

Finally, the electric and magnetic fields that arise from charge movement are taken from Maxwell's Equations.
\begin{align}
    \varepsilon_r \varepsilon_0\frac{\partial \mathbf{E}}{\partial t} &= \frac{1}{\mu_0} \nabla \times \mathbf{B} - e(n_i\mathbf{u}_i - n_e\mathbf{u}_e) \\
    \frac{\partial \mathbf{B}}{\partial t} &= -\nabla \times \mathbf{E}
\end{align}
